bot = telebot.TeleBot(TOKEN)

# Главное меню
def main_menu():
    markup = types.ReplyKeyboardMarkup(resize_keyboard=True, row_width=2)
    markup.add("📊 Статистика Я Тигр", "🛣️ Маршрут")
    markup.add("📋 ФОП Отчёт", "📍 Геопозиция")
    markup.add("🚕 Тоха", "❓ Что ты можешь?")
    return markup

@bot.message_handler(commands=['start'])
def start(message):
    bot.send_message(message.chat.id, 
                     "👋 Привет, Руслан! Я твой личный помощник.\n\n"
                     "Чем могу помочь сегодня?", 
                     reply_markup=main_menu())

# Основной обработчик — делает бота разговорчивым
@bot.message_handler(func=lambda message: True)
def handle_message(message):
    text = message.text.lower()
    chat_id = message.chat.id

    # Приветствия и общение
    if any(word in text for word in ["привет", "здравствуй", "эй", "hi", "hello"]):
        bot.send_message(chat_id, "Привет, Руслан! 😊 Как дела? Чем помочь?", reply_markup=main_menu())

    elif "как дела" in text or "как ты" in text:
        bot.send_message(chat_id, "Отлично! Готов помогать 24/7 🔥 А у тебя как? Что нужно сделать?")

    elif "что можешь" in text or "что ты умеешь" in text:
        bot.send_message(chat_id, "Я могу очень много:\n"
                                  "• Статистику по Я Тигр\n"
                                  "• Отчёты по ФОП\n"
                                  "• Маршруты и отели\n"
                                  "• Работать с Тохой (гео, время забора)\n"
                                  "• И многое другое\n\nЧто сейчас нужно?", reply_markup=main_menu())

    # Уже существующие кнопки
    elif "тигр" in text or "статистика" in text:
        bot.send_message(chat_id, "📊 Делаю статистику по Я Тигр за 2026 год...\n(Скоро подключу твои таблицы)")

    elif "маршрут" in text or "лубны" in text:
        bot.send_message(chat_id, "🛣️ Куда строим маршрут? Отправь гео или название места.")

    elif "фоп" in text:
        bot.send_message(chat_id, "📋 Готовлю отчёт по ФОП 3 группы. Что именно нужно?")

    elif "гео" in text or "где я" in text:
        bot.send_message(chat_id, "📍 Отправь свою геопозицию — я её сохраню и смогу использовать.")

    elif "тоха" in text or "водитель" in text:
        bot.send_message(chat_id, "🚕 Что сказать Тохе?\nПример: «Тоха забери меня в 8:00 дома»")

    elif "спасибо" in text or "благодарю" in text:
        bot.send_message(chat_id, "Пожалуйста! Рад помочь 😊")

    else:
        # Универсальный ответ
        bot.send_message(chat_id, f"Принял: «{message.text}»\n\nЧто именно нужно сделать? Можешь говорить обычными словами.", 
                         reply_markup=main_menu())
